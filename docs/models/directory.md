# Directory

Information for the Directory Sync configuration.

## Example Usage

```typescript
import { Directory } from "@vercel/sdk/models/team.js";

let value: Directory = {
  type: "OktaSAML",
  state: "active",
  connectedAt: 1611796915677,
  lastReceivedWebhookEvent: 1611796915677,
  lastSyncedAt: 1611796915677,
};
```
